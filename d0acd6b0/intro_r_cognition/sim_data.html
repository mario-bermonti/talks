)

data_dsf <- sim_design(
    between = between,
    mu = means,
    sd = sds,
    n = 50,
    dv = "digit_span_forward"

Fix impossible values

data <- data |>
    mutate(
        digit_span_forward = if_else(
            digit_span_forward < 2,
            2,
            digit_span_forward
        ),
        digit_span_backward = if_else(
            digit_span_backward < 2,
            2,
            digit_span_backward
        ),
    )
